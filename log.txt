from typing import Any, Dict


def analyze_candidate(
    candidate_source: str, eval_output: str, artifact_info: Dict[str, Any]
) -> Dict[str, float]:
    """
    Analyzes the performance of a candidate using its source, output, and artifacts.

    This analyzer focuses on the impact of introducing SGDRegressor with early stopping,
    examining convergence behavior, model complexity, and weight distribution.
    """
    metrics: Dict[str, float] = {}

    # Initialize all metrics with a default value for robustness
    keys = [
        "final_score",
        "n_iter_actual",
        "stopped_early_flag",
        "convergence_ratio",
        "total_features",
        "pair_feature_ratio",
        "coef_l2_norm",
        "coef_spread_ratio",
        "intercept_magnitude",
        "score_per_iteration",
        "hyperparam_alpha",
        "hyperparam_pair_scale",
    ]
    for key in keys:
        metrics[f"analysis_{key}"] = -1.0

    # 1. Parse final score from evaluation output
    for line in eval_output.splitlines():
        if "Final score:" in line:
            try:
                score_str = line.split(":")[-1].strip()
                metrics["analysis_final_score"] = float(score_str)
                break
            except (ValueError, IndexError):
                pass  # Keep default if parsing fails

    # 2. Extract information from the analysis artifact
    artifact = artifact_info.get("last_analysis_artifact")
    if isinstance(artifact, dict):
        # --- Convergence and Training Metrics ---
        n_iter_actual = artifact.get("n_iter_actual", -1.0)
        metrics["analysis_n_iter_actual"] = float(n_iter_actual)

        # Parse max_iter from source code to create a convergence ratio
        max_iter_val = 5000.0  # Default from the provided code
        match = re.search(r"max_iter\s*=\s*(\d+)", candidate_source)
        if match:
            max_iter_val = float(match.group(1))

        if n_iter_actual > 0:
            metrics["analysis_stopped_early_flag"] = (
                1.0 if n_iter_actual < max_iter_val else 0.0
            )
            if max_iter_val > 0:
                metrics["analysis_convergence_ratio"] = n_iter_actual / max_iter_val

        # Calculate score efficiency
        final_score = metrics["analysis_final_score"]
        if n_iter_actual > 0 and final_score != -1.0:
            metrics["analysis_score_per_iteration"] = final_score / n_iter_actual

        # --- Feature Engineering Metrics ---
        num_single = artifact.get("num_single_features", 0.0)
        num_pair = artifact.get("num_pair_features", 0.0)
        total_features = num_single + num_pair
        metrics["analysis_total_features"] = float(total_features)
        if total_features > 0:
            metrics["analysis_pair_feature_ratio"] = float(num_pair / total_features)
        else:
            metrics["analysis_pair_feature_ratio"] = 0.0

        # --- Model Parameter Health Metrics ---
        metrics["analysis_coef_l2_norm"] = float(artifact.get("coef_l2_norm", -1.0))
        coef_max_abs = artifact.get("coef_max_abs", 0.0)
        coef_mean_abs = artifact.get("coef_mean_abs", 0.0)
        if coef_mean_abs > 1e-9:
            metrics["analysis_coef_spread_ratio"] = float(coef_max_abs / coef_mean_abs)
        else:
            metrics["analysis_coef_spread_ratio"] = 0.0
        
        intercept = artifact.get("intercept", 0.0)
        metrics["analysis_intercept_magnitude"] = abs(float(intercept))

        # --- Hyperparameter Tracking ---
        metrics["analysis_hyperparam_alpha"] = float(artifact.get("alpha", -1.0))
        metrics["analysis_hyperparam_pair_scale"] = float(artifact.get("pair_scale", -1.0))

    return metrics
