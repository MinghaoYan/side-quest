The key changes are:
1.  **Target Standardization**: The training fitness values are normalized to have a mean of 0 and a standard deviation of 1. This prevents datasets with large-scale fitness values from causing training instability. The model's predictions are then de-standardized to return them to the original scale.
2.  **Gradient Clipping**: `torch.nn.utils.clip_grad_norm_` is applied during training. This function caps the norm of the gradients, preventing them from growing uncontrollably (exploding gradients), a common problem that leads to training failure. A `max_norm` of 1.0 is used.
3.  **Preserving Mutation Order**: Based on the analysis of the prior run, the `sorted()` function call was removed from the mutation parser. This allows the Recurrent Neural Network (RNN) to learn from the sequence order of mutations, which was the original intent of using this architecture.
4.  **Enhanced Artifacts**: The `_PACEVOLVE_LAST_ANALYSIS_ARTIFACT` dictionary is updated to include details about the new stabilization techniques, such as `target_standardization`, `grad_clip_norm`, and the measured mean and standard deviation of the training data.

def fit_and_predict(train_df: pd.DataFrame, test_df: pd.DataFrame, dataset_context: dict) -> np.ndarray:
    """
    Fits a Recurrent Neural Network (RNN) model on low-order mutant data and predicts fitness for high-order mutants.

    This implementation introduces two key stability improvements:
    1.  Target Standardization: Fitness values in the training set are standardized (z-score normalization)
        to have a mean of 0 and a standard deviation of 1. This helps stabilize training across datasets
        with different fitness scales. Predictions are then de-standardized back to the original scale.
    2.  Gradient Clipping: During training, the gradients are clipped to a maximum norm to prevent them
        from exploding, which is a common issue in RNNs, especially with high learning rates.

    The model architecture consists of an embedding layer, a GRU layer, and a final linear layer for regression.
    Mutations are parsed and tokenized into integer sequences. Unlike the previous version, mutations are NOT sorted,
    preserving the original order information from the 'mutant' string to allow the model to capture sequential effects.
    """
    del dataset_context
    global _PACEVOLVE_LAST_ANALYSIS_ARTIFACT

    # --- Helper Functions and Classes (defined inside to be self-contained) ---

    def _parse_mutation(mutation: str) -> List[str]:
        """
        Parses a mutation string into a list of single mutations, preserving order.
        This version does NOT sort the mutations to allow the RNN to model sequential effects.
        """
        text = str(mutation).strip()
        if not text or text == "WT" or text.lower() == "nan":
            return []
        # Preserve order of mutations as they appear in the string
        tokens = [token for token in text.replace(":", "/").split("/") if token and token != "WT"]
        return tokens

    class RNNModel(nn.Module):
        """A simple GRU-based model for sequence regression."""

        def __init__(self, vocab_size: int, embedding_dim: int, hidden_dim: int, n_layers: int, dropout: float):
            super().__init__()
            self.embedding = nn.Embedding(vocab_size, embedding_dim, padding_idx=0)
            self.rnn = nn.GRU(
                embedding_dim,
                hidden_dim,
                n_layers,
                batch_first=True,
                dropout=dropout if n_layers > 1 else 0,
            )
            self.fc = nn.Linear(hidden_dim, 1)
            self.dropout_layer = nn.Dropout(dropout)

        def forward(self, text: torch.Tensor) -> torch.Tensor:
            embedded = self.dropout_layer(self.embedding(text))
            _, hidden = self.rnn(embedded)
            # Use the hidden state from the last layer as the sequence representation
            hidden_last_layer = self.dropout_layer(hidden[-1, :, :])
            output = self.fc(hidden_last_layer)
            return output.squeeze(1)

    # --- 1. Data Preparation & Vocabulary ---
    train_parsed = [_parse_mutation(m) for m in train_df["mutant"]]
    test_parsed = [_parse_mutation(m) for m in test_df["mutant"]]

    # Build vocabulary from all mutations present in train and test sets to avoid unknown tokens
    all_single_mutations = sorted(list(set(m for seq in train_parsed + test_parsed for m in seq)))

    mut_to_idx = {mut: i + 1 for i, mut in enumerate(all_single_mutations)}

    if max_len == 0:

        return np.full(len(test_df), mean_fitness)

    def sequences_to_tensor(parsed_seqs: List[List[str]], max_len: int, mut_to_idx: Dict[str, int]) -> torch.Tensor:
        """Converts lists of parsed mutations to a padded tensor of indices."""
        seqs_as_indices = [[mut_to_idx.get(m, 0) for m in seq] for seq in parsed_seqs]

        padded_seqs = np.zeros((len(seqs_as_indices), max_len), dtype=np.int64)
        for i, seq in enumerate(seqs_as_indices):
            seq_len = len(seq)
            if seq_len > 0:
                padded_seqs[i, :seq_len] = seq[:seq_len]

        return torch.from_numpy(padded_seqs)

    X_train_tensor = sequences_to_tensor(train_parsed, max_len, mut_to_idx)
    y_train_tensor = torch.tensor(train_df["fitness"].values, dtype=torch.float32)
    X_test_tensor = sequences_to_tensor(test_parsed, max_len, mut_to_idx)

    # --- 2. Target Standardization ---
    y_train_mean = y_train_tensor.mean()
    y_train_std = y_train_tensor.std()

    # Add a small epsilon to prevent division by zero if all fitness values are the same
    epsilon = 1e-8
    if y_train_std < epsilon:
        y_train_std = torch.tensor(1.0)  # Avoid scaling if std is zero, just center

    y_train_tensor_scaled = (y_train_tensor - y_train_mean) / y_train_std

    # --- 3. Hyperparameters & Model Setup ---
    EMBEDDING_DIM = 16
    HIDDEN_DIM = 32
    N_LAYERS = 1
    DROPOUT = 0.3
    LEARNING_RATE = 0.01
    EPOCHS = 35
    BATCH_SIZE = 32
    GRAD_CLIP_NORM = 1.0  # Gradient clipping value

    model = RNNModel(
        vocab_size=vocab_size,
        embedding_dim=EMBEDDING_DIM,
        hidden_dim=HIDDEN_DIM,
        n_layers=N_LAYERS,
        dropout=DROPOUT,
    )

    optimizer = optim.Adam(model.parameters(), lr=LEARNING_RATE)
    criterion = nn.MSELoss()

    # --- 4. Training ---
    # Use standardized targets for training
    train_data = TensorDataset(X_train_tensor, y_train_tensor_scaled)
    train_loader = DataLoader(train_data, shuffle=True, batch_size=BATCH_SIZE)

    model.train()
    final_loss = 0.0
    for _ in range(EPOCHS):
        epoch_loss = 0.0
        for seqs, labels in train_loader:
            optimizer.zero_grad()
            predictions = model(seqs)
            loss = criterion(predictions, labels)
            loss.backward()
            # Apply gradient clipping to prevent exploding gradients
            torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=GRAD_CLIP_NORM)
            optimizer.step()
            epoch_loss += loss.item()
        final_loss = epoch_loss / len(train_loader) if train_loader else 0.0

    # --- 5. Prediction ---
    model.eval()
    with torch.no_grad():
        test_data = TensorDataset(X_test_tensor)
        test_loader = DataLoader(test_data, batch_size=BATCH_SIZE * 4, shuffle=False)

        all_preds = []
        for (seqs,) in test_loader:
            preds = model(seqs)
            all_preds.append(preds.cpu().numpy())

        predictions_np_scaled = np.concatenate(all_preds) if all_preds else np.array([])

    # --- 6. De-standardize Predictions ---
    # Convert predictions back to the original fitness scale
    predictions_np = (predictions_np_scaled * y_train_std.item()) + y_train_mean.item()

    # --- 7. Analysis Artifact ---
    _PACEVOLVE_LAST_ANALYSIS_ARTIFACT = {
        "model_type": "RNN_GRU_Stabilized",
        "embedding_dim": float(EMBEDDING_DIM),
        "hidden_dim": float(HIDDEN_DIM),
        "n_layers": float(N_LAYERS),
        "dropout": float(DROPOUT),
        "lr": float(LEARNING_RATE),
        "epochs": float(EPOCHS),
        "batch_size": float(BATCH_SIZE),
        "vocab_size": float(vocab_size),
        "max_seq_len": float(max_len),
        "final_train_mse_loss": float(final_loss),
        "target_standardization": True,
        "grad_clip_norm": float(GRAD_CLIP_NORM),
        "train_fitness_mean": float(y_train_mean.item()),
        "train_fitness_std": float(y_train_std.item()),
    }

    return predictions_np

def analyze_candidate(candidate_source: str, eval_output: str, artifact_info: dict[str, object]) -> dict[str, float]:
    """
    Analyzes the performance of a stabilized RNN candidate, focusing on training stability,
    performance consistency, and the impact of the implemented stabilization techniques.
    """
    metrics = {
        "analysis_error": 1.0,  # Default to 1.0, set to 0.0 on success
    }
    
    try:
        task_artifact = artifact_info.get("task_artifact", {})
        if not task_artifact:
            # Fallback for older schema
            task_artifact = artifact_info.get("structured_artifact", {}).get("payload", {})

        per_dataset_results = task_artifact.get("datasets", [])
        if not per_dataset_results:
            return metrics  # Not enough data to analyze

        losses = []
        scores = []
        train_stds = []
        pred_stds = []
        vocab_sizes = []
        max_lens = []
        d7pm05_recovery_score = -1.0

        for result in per_dataset_results:
            artifact = result.get("candidate_artifact", {})
            perf_metrics = result.get("metrics", {})
            pred_summary = result.get("prediction_summary", {})

            if "final_train_mse_loss" in artifact and "combined_score" in perf_metrics:
                losses.append(artifact["final_train_mse_loss"])
                scores.append(perf_metrics["combined_score"])
                train_stds.append(artifact.get("train_fitness_std", 0.0))
                pred_stds.append(pred_summary.get("prediction_std", 0.0))
                vocab_sizes.append(artifact.get("vocab_size", 0.0))
                max_lens.append(artifact.get("max_seq_len", 0.0))

                if result.get("dataset_id") == "D7PM05_CLYGR_Somermeyer_2022":
                    d7pm05_recovery_score = perf_metrics["combined_score"]

        if not losses:
            return metrics # Not enough valid data points

        # 1. Training stability check
        # Checks if any run had an unusually high final loss on the standardized scale.
        # A loss > 10.0 for z-scored data would indicate a major training failure.
        # 1.0 means stable, 0.0 means unstable.
        max_loss_threshold = 10.0
        is_stable = all(loss < max_loss_threshold for loss in losses)
        metrics["analysis_training_stabilized"] = 1.0 if is_stable else 0.0

        # 2. Average standardized training loss
        # Measures the average final MSE across all datasets. Lower is better.
        metrics["analysis_avg_standardized_loss"] = sum(losses) / len(losses)

        # 3. Consistency of training loss
        # Measures how consistently the model trains across different datasets.
        # Lower standard deviation is better.
        mean_loss = metrics["analysis_avg_standardized_loss"]
        loss_variance = sum((x - mean_loss) ** 2 for x in losses) / len(losses)
        metrics["analysis_std_standardized_loss"] = math.sqrt(loss_variance)

        # 4. Prediction collapse metric
        # Checks if the model's predictions have a reasonable dynamic range compared to the training data.
        # A value near 0 suggests collapse; a value near 1 is ideal.
        normalized_pred_stds = []
        for p_std, t_std in zip(pred_stds, train_stds):
            if t_std > 1e-6:
                normalized_pred_stds.append(p_std / t_std)
        if normalized_pred_stds:
            metrics["analysis_avg_normalized_prediction_std"] = sum(normalized_pred_stds) / len(normalized_pred_stds)
        else:
            metrics["analysis_avg_normalized_prediction_std"] = 0.0
            
        # 5. Fitness dynamic range across datasets
        # Quantifies the variation in fitness scales, justifying the need for standardization.
        # Higher values indicate that standardization was critical.
        valid_stds = [s for s in train_stds if s > 1e-9]
        if len(valid_stds) > 1:
            max_std = max(valid_stds)
            min_std = min(valid_stds)
            metrics["analysis_fitness_dynamic_range_log10"] = math.log10(max_std / min_std)
        else:
            metrics["analysis_fitness_dynamic_range_log10"] = 0.0

        # 6. Performance consistency across datasets
        # A lower standard deviation in scores indicates a more robust and generalizable model.
        mean_score = sum(scores) / len(scores)
        score_variance = sum((x - mean_score) ** 2 for x in scores) / len(scores)
        metrics["analysis_performance_consistency_std"] = math.sqrt(score_variance)

        # 7. Worst-case performance
        # The minimum score on any single dataset highlights the model's primary weakness.
        metrics["analysis_worst_case_performance"] = min(scores)

        # 8. Recovery on previously failed dataset
        # Directly measures the combined score on the dataset that previously failed catastrophically.
        metrics["analysis_d7pm05_recovery_score"] = d7pm05_recovery_score
        
        # 9. Average vocabulary size as a proxy for feature space complexity
        if vocab_sizes:

        # 10. Indication of whether the stabilization was effective

        metrics["analysis_stabilization_effective"] = 1.0 if d7pm05_recovery_score > 0.1 else 0.0

        metrics["analysis_error"] = 0.0

    except (KeyError, IndexError, TypeError, ValueError, ZeroDivisionError) as e:
        # An error occurred during analysis, return default metrics
        # To aid debugging, can log the error type, but for the metric it's just 1.0
        metrics["analysis_error"] = 1.0

def fit_and_predict(train_df: pd.DataFrame, test_df: pd.DataFrame, dataset_context: dict) -> np.ndarray:
    """
    Fits a Recurrent Neural Network (RNN) model on low-order mutant data and predicts fitness for high-order mutants.

    This implementation introduces two key stability improvements:
    1.  Target Standardization: Fitness values in the training set are standardized (z-score normalization)
        to have a mean of 0 and a standard deviation of 1. This helps stabilize training across datasets
        with different fitness scales. Predictions are then de-standardized back to the original scale.
    2.  Gradient Clipping: During training, the gradients are clipped to a maximum norm to prevent them
        from exploding, which is a common issue in RNNs, especially with high learning rates.

    The model architecture consists of an embedding layer, a GRU layer, and a final linear layer for regression.
    Mutations are parsed and tokenized into integer sequences. Unlike the previous version, mutations are NOT sorted,
    preserving the original order information from the 'mutant' string to allow the model to capture sequential effects.
    """
    del dataset_context
    global _PACEVOLVE_LAST_ANALYSIS_ARTIFACT

    # --- Helper Functions and Classes (defined inside to be self-contained) ---

    def _parse_mutation(mutation: str) -> List[str]:
        """
        Parses a mutation string into a list of single mutations, preserving order.
        This version does NOT sort the mutations to allow the RNN to model sequential effects.
        """
        text = str(mutation).strip()
        if not text or text == "WT" or text.lower() == "nan":
            return []
        # Preserve order of mutations as they appear in the string
        tokens = [token for token in text.replace(":", "/").split("/") if token and token != "WT"]
        return tokens

    class RNNModel(nn.Module):
        """A simple GRU-based model for sequence regression."""

        def __init__(self, vocab_size: int, embedding_dim: int, hidden_dim: int, n_layers: int, dropout: float):
            super().__init__()
            self.embedding = nn.Embedding(vocab_size, embedding_dim, padding_idx=0)
            self.rnn = nn.GRU(
                embedding_dim,
                hidden_dim,
                n_layers,
                batch_first=True,
                dropout=dropout if n_layers > 1 else 0,
            )
            self.fc = nn.Linear(hidden_dim, 1)
            self.dropout_layer = nn.Dropout(dropout)

        def forward(self, text: torch.Tensor) -> torch.Tensor:
            embedded = self.dropout_layer(self.embedding(text))
            _, hidden = self.rnn(embedded)
            # Use the hidden state from the last layer as the sequence representation
            hidden_last_layer = self.dropout_layer(hidden[-1, :, :])
            output = self.fc(hidden_last_layer)
            return output.squeeze(1)

    # --- 1. Data Preparation & Vocabulary ---
    train_parsed = [_parse_mutation(m) for m in train_df["mutant"]]
    test_parsed = [_parse_mutation(m) for m in test_df["mutant"]]
